# Step 1: browse all open markets publicly (no credentials needed)
curl "https://external-api.kalshi.com/trade-api/v2/markets?status=open&limit=200" \
  | python3 -c "
import sys,json
for m in json.load(sys.stdin)['markets']:
    print(m['event_ticker'], m['ticker'], m.get('volume_24h','0'))
" | sort -k3 -rn | head -30   # top 30 by 24h volume
